{
    fun counter val -> -> val := val + 1;

    fun fact x -> if x = 0 then 1 else x * fact(x - 1);

    fac := local f := x, res ->
        if x = 0 then {
            res
        } else {
            f(x - 1, x * res)
        }
    in x -> f(x, 1);

    fun map f, l ->
        if l = nil then nil
        else cons(f(head(l)), map(f, tail(l)))
}
